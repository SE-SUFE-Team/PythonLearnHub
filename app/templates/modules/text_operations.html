📝 Python 文本文件操作指南

掌握Python中文本文件的读写、处理和操作技巧

基础文件操作

1. 一次性读取全部内容
# 方法1: 使用 read()
with open('file.txt', 'r', encoding='utf-8') as file:
    content = file.read()
    print(content)

# 方法2: 使用 readlines()
with open('file.txt', 'r', encoding='utf-8') as file:
    lines = file.readlines()
    for line in lines:
        print(line.strip())  # strip() 去除换行符
2. 逐行读取（推荐用于大文件）
# 逐行读取，内存友好
with open('large_file.txt', 'r', encoding='utf-8') as file:
    for line in file:
        # 处理每一行
        print(line.strip())

# 使用 readline() 手动控制
with open('file.txt', 'r', encoding='utf-8') as file:
    line = file.readline()
    while line:
        print(line.strip())
        line = file.readline()
提示: 使用 with 语句可以自动关闭文件，即使出现异常也能确保资源被正确释放。

1. 覆盖写入
# 覆盖模式 'w' - 会清空原文件内容
with open('output.txt', 'w', encoding='utf-8') as file:
    file.write('Hello, World!\n')
    file.write('这是第二行\n')

# 写入多行
lines = ['第一行\n', '第二行\n', '第三行\n']
with open('output.txt', 'w', encoding='utf-8') as file:
    file.writelines(lines)
2. 追加写入
# 追加模式 'a' - 在文件末尾添加内容
with open('log.txt', 'a', encoding='utf-8') as file:
    file.write('新的日志条目\n')

# 追加多行
new_lines = ['追加行1\n', '追加行2\n']
with open('log.txt', 'a', encoding='utf-8') as file:
    file.writelines(new_lines)
注意: 'w' 模式会清空文件原有内容，'a' 模式会在文件末尾追加内容。

常见编码格式
# UTF-8 编码（推荐）
with open('utf8_file.txt', 'r', encoding='utf-8') as file:
    content = file.read()

# GBK 编码（中文Windows系统常用）

# 自动检测编码
import chardet

def detect_encoding(file_path):
    with open(file_path, 'rb') as file:
        raw_data = file.read()
        result = chardet.detect(raw_data)
        return result['encoding']

# 使用检测到的编码读取文件
encoding = detect_encoding('unknown_encoding.txt')
with open('unknown_encoding.txt', 'r', encoding=encoding) as file:
    content = file.read()
编码转换
# 将GBK文件转换为UTF-8
def convert_encoding(input_file, output_file, from_encoding='gbk', to_encoding='utf-8'):
    with open(input_file, 'r', encoding=from_encoding) as infile:
        content = infile.read()
    
    with open(output_file, 'w', encoding=to_encoding) as outfile:
        outfile.write(content)

# 使用示例
convert_encoding('gbk_file.txt', 'utf8_file.txt')
最佳实践: 统一使用 UTF-8 编码，它支持所有Unicode字符且兼容性最好。

高级文本处理技巧

分块读取大文件
# 分块读取，避免内存溢出
def read_large_file_chunks(file_path, chunk_size=1024):
    with open(file_path, 'r', encoding='utf-8') as file:
        while True:
            chunk = file.read(chunk_size)
            if not chunk:
                break
            yield chunk

# 使用生成器处理大文件
for chunk in read_large_file_chunks('very_large_file.txt'):
    # 处理每个块
    process_chunk(chunk)

# 逐行处理大文件（推荐）
def process_large_file_lines(file_path):
    with open(file_path, 'r', encoding='utf-8') as file:
        for line_number, line in enumerate(file, 1):
            # 处理每一行
            processed_line = line.strip().upper()
            print(f"Line {line_number}: {processed_line}")
内存友好的文件操作
# 统计大文件行数（不加载到内存）
def count_lines(file_path):
    count = 0
    with open(file_path, 'r', encoding='utf-8') as file:
        for line in file:
            count += 1
    return count

# 搜索大文件中的特定内容
def search_in_large_file(file_path, search_term):
    matches = []
    with open(file_path, 'r', encoding='utf-8') as file:
        for line_number, line in enumerate(file, 1):
            if search_term in line:
                matches.append((line_number, line.strip()))
    return matches

常见文本清理操作
import re
import string

def clean_text(text):
    """综合文本清理函数"""
    # 1. 去除首尾空白
    text = text.strip()
    
    # 2. 统一换行符
    text = text.replace('\r\n', '\n').replace('\r', '\n')
    
    # 3. 去除多余空白字符
    text = re.sub(r'\s+', ' ', text)
    
    # 4. 去除特殊字符（可选）
    text = re.sub(r'[^\w\s\u4e00-\u9fff]', '', text)
    
    return text

# 文本标准化
def normalize_text(text):
    """文本标准化处理"""
    # 转换为小写
    text = text.lower()
    
    # 去除标点符号
    text = text.translate(str.maketrans('', '', string.punctuation))
    
    # 去除数字（可选）
    text = re.sub(r'\d+', '', text)
    
    return text.strip()

# 处理文件示例
def clean_text_file(input_file, output_file):
    with open(input_file, 'r', encoding='utf-8') as infile:
        with open(output_file, 'w', encoding='utf-8') as outfile:
            for line in infile:
                cleaned_line = clean_text(line)
                if cleaned_line:  # 跳过空行
                    outfile.write(cleaned_line + '\n')
文本统计分析
# 文本统计功能
def analyze_text_file(file_path):
    """分析文本文件的基本统计信息"""
    stats = {
        'total_lines': 0,
        'total_words': 0,
        'total_chars': 0,
        'empty_lines': 0,
        'word_frequency': {}
    }
    
    with open(file_path, 'r', encoding='utf-8') as file:
        for line in file:
            stats['total_lines'] += 1
            
            if line.strip() == '':
                stats['empty_lines'] += 1
                continue
            
            # 统计字符
            stats['total_chars'] += len(line)
            
            # 统计单词
            words = line.strip().split()
            stats['total_words'] += len(words)
            
            # 词频统计
            for word in words:
                word = word.lower().strip(string.punctuation)
                if word:
                    stats['word_frequency'][word] = stats['word_frequency'].get(word, 0) + 1
    
    return stats

# 使用示例
stats = analyze_text_file('sample.txt')
print(f"总行数: {stats['total_lines']}")
print(f"总词数: {stats['total_words']}")
print(f"总字符数: {stats['total_chars']}")

# 显示最常见的10个词
top_words = sorted(stats['word_frequency'].items(), key=lambda x: x[1], reverse=True)[:10]
print("最常见的词:")
for word, freq in top_words:
    print(f"  {word}: {freq}")

实用代码示例

CSV文件处理 
import csv

# 读取CSV文件
def read_csv_file(file_path):
    data = []
    with open(file_path, 'r', encoding='utf-8') as file:
        reader = csv.DictReader(file)
        for row in reader:
            data.append(row)
    return data

# 写入CSV文件
def write_csv_file(file_path, data, fieldnames):
    with open(file_path, 'w', newline='', encoding='utf-8') as file:
        writer = csv.DictWriter(file, fieldnames=fieldnames)
        writer.writeheader()
        writer.writerows(data)
JSON文件处理 
import json

# 读取JSON文件
def read_json_file(file_path):
    with open(file_path, 'r', encoding='utf-8') as file:
        data = json.load(file)
    return data

# 写入JSON文件
def write_json_file(file_path, data):
    with open(file_path, 'w', encoding='utf-8') as file:
        json.dump(data, file, ensure_ascii=False, indent=4)

# 美化JSON输出
def pretty_print_json(data):
    return json.dumps(data, ensure_ascii=False, indent=2)

最佳实践总结

推荐做法
  • 始终使用 with 语句操作文件
  • 明确指定文件编码（建议UTF-8）
  • 处理大文件时使用逐行读取
  • 添加异常处理机制
  • 使用有意义的变量名
避免的做法
  • 忘记关闭文件句柄
  • 一次性加载超大文件到内存
  • 忽略文件编码问题
  • 不处理文件操作异常
  • 硬编码文件路径
